    // ==================== 雨滴系统 ====================
    class RainSystem {
      constructor(canvas) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.drops = [];
        this.width = window.innerWidth;
        this.height = window.innerHeight;
        this.canvas.width = this.width;
        this.canvas.height = this.height;
        this.init();
        this.animate();
        
        window.addEventListener('resize', () => {
          this.width = window.innerWidth;
          this.height = window.innerHeight;
          this.canvas.width = this.width;
          this.canvas.height = this.height;
        });
      }
      
      init() {
        for (let i = 0; i < 200; i++) {
          this.drops.push({
            x: Math.random() * this.width,
            y: Math.random() * this.height,
            length: Math.random() * 20 + 10,
            speed: Math.random() * 10 + 5,
            opacity: Math.random() * 0.5 + 0.2
          });
        }
      }
      
      animate() {
        this.ctx.clearRect(0, 0, this.width, this.height);
        
        const count = Math.floor(AppState.rainIntensity * 2);
        
        this.ctx.strokeStyle = 'rgba(200, 220, 255, 0.6)';
        this.ctx.lineWidth = 1;
        
        for (let i = 0; i < Math.min(count, this.drops.length); i++) {
          const drop = this.drops[i];
          
          this.ctx.beginPath();
          this.ctx.moveTo(drop.x, drop.y);
          this.ctx.lineTo(drop.x, drop.y + drop.length);
          this.ctx.globalAlpha = drop.opacity;
          this.ctx.stroke();
          
          drop.y += drop.speed;
          if (drop.y > this.height) {
            drop.y = -drop.length;
            drop.x = Math.random() * this.width;
          }
        }
        
        requestAnimationFrame(() => this.animate());
      }
    }

    let handbookCtx = null;
    let isDrawing = false;
    let currentTool = 'pen';
    let hbCanvas = null;

    function initHandbook() {
      hbCanvas = document.getElementById('handbookCanvas');
      handbookCtx = hbCanvas.getContext('2d');
      
      const resizeCanvas = () => {
        const container = document.getElementById('handbookPaper');
        hbCanvas.width = container.clientWidth;
        hbCanvas.height = container.clientHeight;
        // 重绘当前页面
        if (AppState.currentHandbookPage?.canvasData) {
          const img = new Image();
          img.onload = () => handbookCtx.drawImage(img, 0, 0);
          img.src = AppState.currentHandbookPage.canvasData;
        }
      };
      resizeCanvas();
      window.addEventListener('resize', resizeCanvas);
      
      // 绘图事件
      hbCanvas.addEventListener('mousedown', startDrawing);
      hbCanvas.addEventListener('mousemove', draw);
      hbCanvas.addEventListener('mouseup', stopDrawing);
      hbCanvas.addEventListener('mouseout', stopDrawing);
      
      // 触摸支持
      hbCanvas.addEventListener('touchstart', (e) => {
        e.preventDefault();
        const touch = e.touches[0];
        const rect = hbCanvas.getBoundingClientRect();
        startDrawing({ clientX: touch.clientX, clientY: touch.clientY });
      });
      hbCanvas.addEventListener('touchmove', (e) => {
        e.preventDefault();
        const touch = e.touches[0];
        draw({ clientX: touch.clientX, clientY: touch.clientY });
      });
      hbCanvas.addEventListener('touchend', stopDrawing);
      
      renderHandbookList();
    }

    function startDrawing(e) {
      if (currentTool !== 'pen') return;
      isDrawing = true;
      const rect = hbCanvas.getBoundingClientRect();
      handbookCtx.beginPath();
      handbookCtx.moveTo(e.clientX - rect.left, e.clientY - rect.top);
    }

    function draw(e) {
      if (!isDrawing || currentTool !== 'pen') return;
      const rect = hbCanvas.getBoundingClientRect();
      handbookCtx.lineWidth = document.getElementById('hbSize').value;
      handbookCtx.strokeStyle = document.getElementById('hbColor').value;
      handbookCtx.lineCap = 'round';
      handbookCtx.lineJoin = 'round';
      handbookCtx.lineTo(e.clientX - rect.left, e.clientY - rect.top);
      handbookCtx.stroke();
    }

    function stopDrawing() {
      isDrawing = false;
      handbookCtx.beginPath();
    }

    function formatPomoTime(seconds) {
      const mins = Math.floor(seconds / 60);
      return `${mins}分钟`;
    }

    function startPomodoro() {
      if (!pomoCurrentTask) {
        showToast('请先选择一个任务');
        return;
      }
      if (pomoIsRunning) return;
      
      pomoIsRunning = true;
      document.getElementById('pomoStart').classList.add('hidden');
      document.getElementById('pomoPause').classList.remove('hidden');
      
      pomoTimer = setInterval(() => {
        pomoTimeLeft--;
        updatePomoDisplay();
        
        // 记录时间
        const today = new Date().toDateString();
        if (!AppState.pomodoro.records[today]) AppState.pomodoro.records[today] = {};
        if (!AppState.pomodoro.records[today][pomoCurrentTask.id]) {
          AppState.pomodoro.records[today][pomoCurrentTask.id] = 0;
        }
        AppState.pomodoro.records[today][pomoCurrentTask.id]++;
        pomoCurrentTask.totalTime = (pomoCurrentTask.totalTime || 0) + 1;
        
        if (pomoTimeLeft <= 0) {
          stopPomodoro();
          musicSystem.play('whale');
          setTimeout(() => musicSystem.stop(), 5000);
          showToast('🎉 专注完成！');
        }
        
        saveCurrentAccount();
      }, 1000);
    }

    function pausePomodoro() {
      pomoIsRunning = false;
      clearInterval(pomoTimer);
      document.getElementById('pomoStart').classList.remove('hidden');
      document.getElementById('pomoPause').classList.add('hidden');
    }

    function stopPomodoro() {
      pomoIsRunning = false;
      clearInterval(pomoTimer);
      pomoTimeLeft = pomoMode * 60;
      updatePomoDisplay();
      document.getElementById('pomoStart').classList.remove('hidden');
      document.getElementById('pomoPause').classList.add('hidden');
    }

    function updatePomoDisplay() {
      const mins = Math.floor(pomoTimeLeft / 60);
      const secs = pomoTimeLeft % 60;
      document.getElementById('pomodoroTimer').textContent = 
        `${String(mins).padStart(2, '0')}:${String(secs).padStart(2, '0')}`;
    }

    function renderPomoCharts() {
      // 今日扇形图
      const today = new Date().toDateString();
      const todayData = AppState.pomodoro.records[today] || {};
      const tasks = AppState.pomodoro.tasks || [];
      
      const colors = ['#9D2933', '#F3C13A', '#1E3A5F', '#9D6656', '#5E8B6E', '#B8A1C9'];
      const chartData = tasks.map((task, i) => ({
        name: task.name,
        value: (todayData[task.id] || 0) / 60,
        color: colors[i % colors.length]
      })).filter(d => d.value > 0);
      
      const total = chartData.reduce((sum, d) => sum + d.value, 0);
      document.getElementById('todayPomoTotal').textContent = total > 0 ? `今日专注: ${Math.floor(total)}分钟` : '今日暂无专注记录';
      
      // 绘制扇形图
      const canvas = document.getElementById('todayPomoChart');
      const ctx = canvas.getContext('2d');
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      
      if (chartData.length === 0) {
        ctx.fillStyle = 'rgba(255,255,255,0.1)';
        ctx.beginPath();
        ctx.arc(100, 100, 80, 0, Math.PI * 2);
        ctx.fill();
        ctx.fillStyle = 'rgba(255,255,255,0.5)';
        ctx.font = '12px sans-serif';
        ctx.textAlign = 'center';
        ctx.fillText('暂无数据', 100, 105);
        return;
      }
      
      let startAngle = -Math.PI / 2;
      chartData.forEach(data => {
        const angle = (data.value / total) * Math.PI * 2;
        ctx.fillStyle = data.color;
        ctx.beginPath();
        ctx.moveTo(100, 100);
        ctx.arc(100, 100, 80, startAngle, startAngle + angle);
        ctx.closePath();
        ctx.fill();
        startAngle += angle;
      });
      
      // 中心空白
      ctx.fillStyle = '#0a0f1a';
      ctx.beginPath();
      ctx.arc(100, 100, 50, 0, Math.PI * 2);
      ctx.fill();
      
      // 周统计条形图
      renderWeekChart();
    }

    function renderWeekChart() {
      const canvas = document.getElementById('weekPomoChart');
      if (!canvas) return;
      const ctx = canvas.getContext('2d');
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      
      const days = [];
      const data = [];
      for (let i = 6; i >= 0; i--) {
        const date = new Date();
        date.setDate(date.getDate() - i);
        const dateStr = date.toDateString();
        days.push(date.toLocaleDateString('zh-CN', { weekday: 'short' }));
        
        const records = AppState.pomodoro.records[dateStr] || {};
        const total = Object.values(records).reduce((sum, v) => sum + v, 0);
        data.push(total / 60);
      }
      
      const max = Math.max(...data, 1);
      const barWidth = 30;
      const gap = 15;
      const startX = 20;
      
      data.forEach((value, i) => {
        const height = (value / max) * 80;
        const x = startX + i * (barWidth + gap);
        const y = 100 - height;
        
        const gradient = ctx.createLinearGradient(0, y, 0, 100);
        gradient.addColorStop(0, '#9D2933');
        gradient.addColorStop(1, '#F3C13A');
        ctx.fillStyle = gradient;
        ctx.fillRect(x, y, barWidth, height);
        
        ctx.fillStyle = 'rgba(255,255,255,0.5)';
        ctx.font = '10px sans-serif';
        ctx.textAlign = 'center';
        ctx.fillText(days[i], x + barWidth / 2, 115);
        
        if (value > 0) {
          ctx.fillText(Math.floor(value) + 'm', x + barWidth / 2, y - 5);
        }
      });
    }

    // ==================== 市场分析 ====================
    function analyzeMarketData() {
      const input = document.getElementById('marketDataInput').value.trim();
      if (!input) {
        showToast('请输入数据');
        return;
      }
      
      const lines = input.split('\n');
      const books = [];
      const genres = ['都市', '玄幻', '仙侠', '科幻', '悬疑', '古言', '现言', '重生', '系统', '无限流', '种田', '基建', '快穿', '穿书', '末世', '星际', '西幻', '东方', '历史', '军事', '游戏', '体育', '轻小说'];
      const tags = ['甜宠', '爽文', '虐渣', '复仇', '先婚后爱', '权谋', '穿越', '快穿', '囤货', '基建', '团宠', '马甲', '打脸', '逆袭', '双强', '破镜重圆', '暗恋', '青梅竹马', '年上', '年下', '强强', '治愈', '致郁', '轻松', '正剧', '悲剧', '喜剧'];
      
      lines.forEach(line => {
        const match = line.match(/[《"']([^《"']+)[《"']/);
        if (match) {
          books.push({ title: match[1], fullLine: line });
        }
      });
      
      if (books.length === 0) {
        showToast('未能识别到有效数据');
        return;
      }
      
      // 分析题材
      const foundGenres = genres.filter(g => input.includes(g));
      // 分析标签
      const foundTags = tags.filter(t => input.includes(t));
      // 书名模式
      const titlePatterns = books.slice(0, 5).map(b => b.title);
      
      // 显示结果
      document.getElementById('marketAnalysisResult').classList.remove('hidden');
      
      document.getElementById('hotGenres').innerHTML = foundGenres.length > 0 
        ? foundGenres.map(g => `<span class="px-2 py-1 rounded-full text-xs bg-yanzhi/30 text-white border border-yanzhi/50">${g}</span>`).join('')
        : '<span class="text-moon-white/40 text-xs">未识别到热门题材</span>';
      
      document.getElementById('hotTags').innerHTML = foundTags.length > 0
        ? foundTags.map(t => `<span class="px-2 py-1 rounded-full text-xs bg-tenghuang/30 text-white border border-tenghuang/50">${t}</span>`).join('')
        : '<span class="text-moon-white/40 text-xs">未识别到标签</span>';
      
      document.getElementById('titleFormulas').innerHTML = titlePatterns.map(t => `<li class="text-moon-white/70">《${t}》</li>`).join('');
      
      // 生成建议
      let advice = `分析到${books.length}本作品。`;
      if (foundGenres.length > 0) advice += `热门题材：${foundGenres.slice(0, 3).join('、')}。`;
      if (foundTags.length > 0) advice += `推荐标签：${foundTags.slice(0, 3).join('、')}。`;
      advice += '建议书名控制在6-10字，前3章设置强冲突点，每章结尾留钩子。';
      
      document.getElementById('writingAdvice').textContent = advice;
    }

    // ==================== 导出功能 ====================
    function exportData(format) {
      if (format === 'txt') {
        if (!AppState.currentProject) {
          showToast('请先选择一个项目');
          return;
        }
        const content = AppState.currentProject.chapters?.map(c => `${c.title}\n\n${c.content}`).join('\n\n---\n\n') || '';
        downloadFile(`${AppState.currentProject.title}.txt`, content);
      } else if (format === 'md') {
        if (!AppState.currentProject) {
          showToast('请先选择一个项目');
          return;
        }
        const content = AppState.currentProject.chapters?.map(c => `# ${c.title}\n\n${c.content}`).join('\n\n---\n\n') || '';
        downloadFile(`${AppState.currentProject.title}.md`, content);
      } else if (format === 'json') {
        const data = JSON.stringify(AppState.accounts[AppState.currentUser], null, 2);
        downloadFile(`rainwindow_backup_${AppState.currentUser}_${new Date().toISOString().split('T')[0]}.json`, data);
      }
    }

    function downloadFile(filename, content) {
      const blob = new Blob([content], { type: 'text/plain;charset=utf-8' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = filename;
      a.click();
      URL.revokeObjectURL(url);
      showToast('导出成功');
    }

    function importBackup(file) {
      const reader = new FileReader();
      reader.onload = (e) => {
        try {
          const data = JSON.parse(e.target.result);
          AppState.accounts[AppState.currentUser] = data;
          saveAccounts();
          // 重新加载
          login(AppState.currentUser);
          showToast('数据恢复成功');
        } catch (err) {
          showToast('文件格式错误');
        }
      };
      reader.readAsText(file);
    }

      document.getElementById('pomoStart').onclick = startPomodoro;
      document.getElementById('pomoPause').onclick = pausePomodoro;
      document.getElementById('pomoStop').onclick = stopPomodoro;
      
      // 番茄钟时间模式
      document.querySelectorAll('.pomo-mode').forEach(btn => {
        btn.onclick = () => {
          document.querySelectorAll('.pomo-mode').forEach(b => {
            b.classList.remove('bg-yanzhi/30', 'text-white', 'border-yanzhi');
            b.classList.add('text-moon-white/60');
          });
          btn.classList.remove('text-moon-white/60');
          btn.classList.add('bg-yanzhi/30', 'text-white', 'border-yanzhi');
          pomoMode = parseInt(btn.dataset.time);
          if (!pomoIsRunning) {
            pomoTimeLeft = pomoMode * 60;
            updatePomoDisplay();
          }
        };
      });
      
      // 手账本工具
      document.getElementById('hbToolPen').onclick = () => {
        currentTool = 'pen';
        document.querySelectorAll('.tool-btn').forEach(b => b.classList.remove('btn-active'));
        document.getElementById('hbToolPen').classList.add('btn-active');
      };
      document.getElementById('hbToolText').onclick = () => {
        currentTool = 'text';
        document.querySelectorAll('.tool-btn').forEach(b => b.classList.remove('btn-active'));
        document.getElementById('hbToolText').classList.add('btn-active');
        document.getElementById('handbookText').focus();
      };
      document.getElementById('hbClear').onclick = () => {
        handbookCtx.clearRect(0, 0, hbCanvas.width, hbCanvas.height);
      };
      document.getElementById('hbSave').onclick = () => {
        if (AppState.currentHandbookPage) {
          AppState.currentHandbookPage.text = document.getElementById('handbookText').value;
          AppState.currentHandbookPage.canvasData = hbCanvas.toDataURL();
          saveCurrentAccount();
          showToast('手账已保存');
        }
      };
      document.getElementById('paperType').onchange = updatePaperType;
      document.getElementById('newHandbookPage').onclick = () => {
        // 保存当前
        if (AppState.currentHandbookPage) {
          AppState.currentHandbookPage.text = document.getElementById('handbookText').value;
          AppState.currentHandbookPage.canvasData = hbCanvas.toDataURL();
        }
        // 新建
        const newPage = {
          id: Date.now(),
          date: new Date().toISOString(),
          text: '',
          canvasData: null,
          paperType: document.getElementById('paperType').value
        };
        AppState.handbook.push(newPage);
        AppState.currentHandbookPage = newPage;
        document.getElementById('handbookText').value = '';
        handbookCtx.clearRect(0, 0, hbCanvas.width, hbCanvas.height);
        saveCurrentAccount();
        renderHandbookList();
        showToast('新页面已创建');
      };
      
      // 写作提示
      let currentPromptType = 'plot';
      document.querySelectorAll('.prompt-tab').forEach(btn => {
        btn.onclick = () => {
          document.querySelectorAll('.prompt-tab').forEach(b => {
            b.classList.remove('bg-rain-blue/20', 'text-rain-blue', 'border-rain-blue/30');
            b.classList.add('text-moon-white/60');
          });
          btn.classList.remove('text-moon-white/60');
          btn.classList.add('bg-rain-blue/20', 'text-rain-blue', 'border-rain-blue/30');
          currentPromptType = btn.dataset.type;
        };
      });
      document.getElementById('generatePrompt').onclick = () => {
        const prompts = AppState.writingPrompts[currentPromptType];
        const prompt = prompts[Math.floor(Math.random() * prompts.length)];
        document.getElementById('promptContent').textContent = prompt;
      };
      
      // 导出
      document.querySelectorAll('.export-btn').forEach(btn => {
        btn.onclick = () => exportData(btn.dataset.format);
      });
      
      // 备份
      document.getElementById('exportBackupBtn').onclick = () => exportData('json');
      document.getElementById('importBackupInput').onchange = (e) => {
        if (e.target.files[0]) {
          importBackup(e.target.files[0]);
        }
      };
      
      // 遮罩点击
      document.getElementById('overlay').onclick = closeAllDrawers;
      
      // 快捷键
      document.addEventListener('keydown', (e) => {
        if (e.ctrlKey || e.metaKey) {
          if (e.key === 's') {
            e.preventDefault();
            document.getElementById('saveBtn').click();
          }
          if (e.key === 'f') {
            e.preventDefault();
            document.getElementById('focusModeBtn').click();
          }
        }
        if (e.key === 'Escape' && AppState.isFocusMode) {
          exitFocusMode();
        }
      });
      
      // 自动加载默认项目
      if (AppState.projects.length > 0) {
        openProject(AppState.projects[0].id);
      }
    }

    // 启动
    document.addEventListener('DOMContentLoaded', () => {
      // 加载已有账号
      const saved = localStorage.getItem('rainWindow_accounts');
      if (saved) {
        AppState.accounts = JSON.parse(saved);
      }
      
      playSplashAnimation();
      initLogin();
    });
